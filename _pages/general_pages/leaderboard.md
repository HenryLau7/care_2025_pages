function parseNum(text) {
  const t = (text || "").toString().trim().replace(/[^\d.eE+-]/g, "");
  const v = parseFloat(t);
  return Number.isNaN(v) ? NaN : v;
}

function sortTableBy(table, colIndex, ascending = true) {
  const tbody = table.tBodies[0];
  const rows = Array.from(tbody.querySelectorAll("tr"));
  rows.sort((r1, r2) => {
    const v1 = parseNum(r1.cells[colIndex]?.textContent);
    const v2 = parseNum(r2.cells[colIndex]?.textContent);
    if (Number.isNaN(v1) && Number.isNaN(v2)) return 0;
    if (Number.isNaN(v1)) return 1;
    if (Number.isNaN(v2)) return -1;
    return ascending ? (v1 - v2) : (v2 - v1);
  });
  rows.forEach(r => tbody.appendChild(r));
}

function ensureColumnLayout(parent) {
  const cs = window.getComputedStyle(parent);
  if (cs.display === "flex" && cs.flexDirection !== "column") {
    parent.style.flexDirection = "column";
    if (!cs.gap || cs.gap === "0px") parent.style.gap = "8px";
    parent.style.alignItems = "stretch";
  }
}

function createBtn(label) {
  const b = document.createElement("button");
  b.type = "button";
  b.textContent = label;
  b.style.cssText = "padding:6px 10px;border:1px solid #ccc;border-radius:6px;background:#f7f7f7;cursor:pointer;";
  return b;
}

function buildWrapper() {
  const wrapper = document.createElement("div");
  wrapper.className = "sort-wrapper";
  wrapper.style.margin = "8px 0";
  wrapper.style.display = "flex";
  wrapper.style.alignItems = "center";
  wrapper.style.flexWrap = "wrap";
  wrapper.style.gap = "8px";
  return wrapper;
}

// =============== 通用初始化器 ===============
function initSortableTables({ tableClass, buttons, defaultKey }) {
  const tables = document.querySelectorAll(`table.${tableClass}`);
  tables.forEach((table) => {
    if (!table || table.dataset.sortInit === "1") return;             // 避免重复初始化
    if (table.previousElementSibling?.classList?.contains("sort-wrapper")) return;

    ensureColumnLayout(table.parentNode);

    const wrapper = buildWrapper();
    const hint = document.createElement("span");
    hint.style.cssText = "color:#666;font-size:0.9em;margin-left:8px;";

    const btnElems = {};
    buttons.forEach(btn => {
      const b = createBtn(btn.label);
      btnElems[btn.key] = b;
      wrapper.appendChild(b);
    });
    wrapper.appendChild(hint);
    table.parentNode.insertBefore(wrapper, table);

    let mode = defaultKey;

    function setButtons() {
      buttons.forEach(btn => {
        const b = btnElems[btn.key];
        const active = mode === btn.key;
        b.disabled = active;
        b.style.opacity = active ? "0.6" : "1";
      });
      const current = buttons.find(b => b.key === mode);
      hint.textContent = `Sorted by ${current.label}`;
    }

    function applySort() {
      const current = buttons.find(b => b.key === mode);
      if (!current) return;
      sortTableBy(table, current.col, current.asc);
      setButtons();
    }

    // 事件绑定
    buttons.forEach(btn => {
      btnElems[btn.key].addEventListener("click", () => {
        if (mode === btn.key) return;
        mode = btn.key;
        applySort();
      });
    });

    // 初始排序
    applySort();
    table.dataset.sortInit = "1";
  });
}
